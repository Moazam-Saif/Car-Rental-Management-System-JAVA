package com.example.carrentalmanagmentsystem;

public class CarCard {

    private String carCardImageSRC;
    private String carCardName;
    private String carCardPrice;

    public String getCarCardImageSRC() {
        return carCardImageSRC;
    }

    public void setCarCardImageSRC(String carCardImageSRC) {
        this.carCardImageSRC = carCardImageSRC;
    }

    public String getCarCardName() {
        return carCardName;
    }

    public void setCarCardName(String carCardName) {
        this.carCardName = carCardName;
    }

    public String getCarCardPrice() {
        return carCardPrice;
    }

    public void setCarCardPrice(String carCardPrice) {
        this.carCardPrice = carCardPrice;
    }
}
